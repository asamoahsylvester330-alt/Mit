function runProcess() { if (!mainWB || !bioWB) return; const btn = document.getElementById('btn-process'); btn.disabled = true; resultBytes = null; document.getElementById('btn-download').style.display = 'none'; document.getElementById('metrics-wrap').style.display = 'none'; document.getElementById('log-box').innerHTML = ''; pb(0); log('Reading main report…'); let rows = sheetToObjects(mainWB, 0); rows = rows.map(r => { const clean = {}; for (const k of Object.keys(r)) clean[k.trim()] = r[k]; return clean; }); const originalRows = rows.length; pb(10); // ───────────────────────────────────────────── // BIO DATA // ───────────────────────────────────────────── log('Reading bio data…'); let bioRows = sheetToObjects(bioWB, 0); bioRows = bioRows.map(r => { const clean = {}; for (const k of Object.keys(r)) clean[k.trim()] = r[k]; return clean; }); const bioMap = {}; for (const r of bioRows) { const key = String(r['MEMBER NO'] || '').trim(); if (key) bioMap[key] = String(r['FULL NAME'] || '').trim(); } pb(20); // ───────────────────────────────────────────── // NAME MATCHING // ───────────────────────────────────────────── log('Matching employee numbers…'); let matched = 0, unmatched = 0; for (const row of rows) { const empNo = String(row['Employee_Number'] || '').trim(); const fullName = bioMap[empNo] || ''; if (fullName) { const parts = fullName.split(/\s+/); row['First_Name'] = parts[0] || ''; row['Last_Name'] = parts[1] || ''; row['Other_Name'] = parts.slice(2).join(' ') || ''; matched++; } else { row['First_Name'] = ''; row['Last_Name'] = ''; row['Other_Name'] = ''; unmatched++; } } pb(30); // ───────────────────────────────────────────── // FILTERS // ───────────────────────────────────────────── log('Filtering invalid rows…'); const BAD_WORDS = ['UNKNOWN','DUMMY','SUSPENSE','OVERPAYMENT','UNDERPAYMENT','MEMBER']; const beforeFilter = rows.length; rows = rows.filter(r => { const txt = Object.values(r).join(' ').toUpperCase(); return !BAD_WORDS.some(w => txt.includes(w)); }); pb(40); // ───────────────────────────────────────────── // BENEFITS FILTER // ───────────────────────────────────────────── const CC = 'Accumulated contributions at the end of current month'; const BC = 'Accrued Benefits at the end of current month'; const beforeBenefit = rows.length; rows = rows.filter(r => { const acc = parseFloat(String(r[CC]).replace(/,/g,'')); const ben = parseFloat(String(r[BC]).replace(/,/g,'')); return !isNaN(acc) && !isNaN(ben) && acc > 0 && ben > 0; }); pb(50); // ───────────────────────────────────────────── // GHANA CARD RULE (NEW FIX) // ───────────────────────────────────────────── log('Cleaning Ghana Card numbers…'); let ghanaFixed = 0; for (const r of rows) { let v = String(r['Ghana_Card_Number'] || '').trim(); if (!v || v === '' || v === '0' || v.toLowerCase() === 'undefined') { r['Ghana_Card_Number'] = '00'; // TEXT FORMAT ghanaFixed++; } else { r['Ghana_Card_Number'] = v; } } pb(58); // ───────────────────────────────────────────── // DOB RULE ENGINE // ───────────────────────────────────────────── log('Applying Date of Birth rules…'); const uploadDate = new Date(); let dobFixed = 0; function parseAnyDate(v) { if (!v || String(v).trim() === '') return null; const d = new Date(v); return isNaN(d.getTime()) ? null : d; } function fmtDate(d) { const mm = String(d.getMonth() + 1).padStart(2, '0'); const dd = String(d.getDate()).padStart(2, '0'); const yy = String(d.getFullYear()); return `${mm}/${dd}/${yy}`; } for (const r of rows) { let dob = parseAnyDate(r['Date_Of_Birth']); let invalid = false; if (!dob) { invalid = true; } else { const age = (uploadDate - dob) / (365.25 * 24 * 60 * 60 * 1000); const year = dob.getFullYear(); if (age > 55 || year >= 2007) { invalid = true; } } if (invalid) { r['Date_Of_Birth'] = '01/01/1998'; dobFixed++; } else { r['Date_Of_Birth'] = fmtDate(dob); } } pb(70); // ───────────────────────────────────────────── // MONTH LAG DATES // ───────────────────────────────────────────── log('Applying month-lag contribution dates…'); const today = new Date(); const lastDayPrevMonth = new Date(today.getFullYear(), today.getMonth(), 0); const lastButOne = new Date(lastDayPrevMonth); lastButOne.setDate(lastDayPrevMonth.getDate() - 1); const fallback = fmtDate(lastButOne); const dateCols = [ 'Contributions_Allocation_Date', 'Contributions_Receipt_Date', 'Contrib Period', 'Original_Date_Earned' ]; for (const col of dateCols) { for (const r of rows) { let d = parseAnyDate(r[col]); r[col] = d ? fmtDate(d) : fallback; } } pb(85); // ───────────────────────────────────────────── // OUTPUT FILE // ───────────────────────────────────────────── log('Generating output file…'); const outWB = XLSX.utils.book_new(); XLSX.utils.book_append_sheet( outWB, XLSX.utils.json_to_sheet(rows), 'Clean_Report' ); const audit = [ ['Metric','Count'], ['Original Rows', originalRows], ['Matched Names', matched], ['Unmatched Names', unmatched], ['Filtered Rows', beforeFilter - beforeBenefit], ['Invalid Benefits Removed', beforeBenefit - rows.length], ['DOB Fixed', dobFixed], ['Ghana Card Fixed', ghanaFixed], ['Final Rows', rows.length] ]; XLSX.utils.book_append_sheet( outWB, XLSX.utils.aoa_to_sheet(audit), 'Audit' ); resultBytes = XLSX.write(outWB, { bookType:'xlsx', type:'array' }); pb(100); log(`Done — ${rows.length.toLocaleString()} rows ready`, 'ok'); document.getElementById('m-orig').textContent = originalRows; document.getElementById('m-matched').textContent = matched; document.getElementById('m-removed').textContent = originalRows - rows.length; document.getElementById('m-final').textContent = rows.length; document.getElementById('metrics-wrap').style.display = 'block'; document.getElementById('btn-download').style.display = 'flex'; btn.style.display = 'none'; setStep(4); }
